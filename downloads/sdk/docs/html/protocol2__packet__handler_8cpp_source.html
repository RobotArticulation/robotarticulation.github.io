200 void Protocol2PacketHandler::addStuffing(uint8_t *packet)
201 {
202  int packet_length_in = MCY_MAKEWORD(packet[PKT_LENGTH_L], packet[PKT_LENGTH_H]);
203  int packet_length_out = packet_length_in;
204 
205  if (packet_length_in < 8) // INSTRUCTION, ADDR_L, ADDR_H, CRC16_L, CRC16_H + FF FF FD
206  return;
207 
208  uint8_t *packet_ptr;
209  uint16_t packet_length_before_crc = packet_length_in - 2;
210  for (uint16_t i = 3; i < packet_length_before_crc; i++)
211  {
212  packet_ptr = &packet[i+PKT_INSTRUCTION-2];
213  if (packet_ptr[0] == 0xFF && packet_ptr[1] == 0xFF && packet_ptr[2] == 0xFD)
214  packet_length_out++;
215  }
216 
217  if (packet_length_in == packet_length_out) // no stuffing required
218  return;
219 
220  uint16_t out_index = packet_length_out + 6 - 2; // last index before crc
221  uint16_t in_index = packet_length_in + 6 - 2; // last index before crc
222  while (out_index != in_index)
223  {
224  if (packet[in_index] == 0xFD && packet[in_index-1] == 0xFF && packet[in_index-2] == 0xFF)
225  {
226  packet[out_index--] = 0xFD; // byte stuffing
227  if (out_index != in_index)
228  {
229  packet[out_index--] = packet[in_index--]; // FD
230  packet[out_index--] = packet[in_index--]; // FF
231  packet[out_index--] = packet[in_index--]; // FF
232  }
233  }
234  else
235  {
236  packet[out_index--] = packet[in_index--];
237  }
238  }
239 
240  packet[PKT_LENGTH_L] = MCY_LOBYTE(packet_length_out);
241  packet[PKT_LENGTH_H] = MCY_HIBYTE(packet_length_out);
242 
243  return;
244 }

246 void Protocol2PacketHandler::removeStuffing(uint8_t *packet)
247 {
248  int i = 0, index = 0;
249  int packet_length_in = MCY_MAKEWORD(packet[PKT_LENGTH_L], packet[PKT_LENGTH_H]);
250  int packet_length_out = packet_length_in;
251 
252  index = PKT_INSTRUCTION;
253  for (i = 0; i < packet_length_in - 2; i++) // except CRC
254  {
255  if (packet[i+PKT_INSTRUCTION] == 0xFD && packet[i+PKT_INSTRUCTION+1] == 0xFD && packet[i+PKT_INSTRUCTION-1] == 0xFF && packet[i+PKT_INSTRUCTION-2] == 0xFF)
256  { // FF FF FD FD
257  packet_length_out--;
258  i++;
259  }
260  packet[index++] = packet[i+PKT_INSTRUCTION];
261  }
262  packet[index++] = packet[PKT_INSTRUCTION+packet_length_in-2];
263  packet[index++] = packet[PKT_INSTRUCTION+packet_length_in-1];
264 
265  packet[PKT_LENGTH_L] = MCY_LOBYTE(packet_length_out);
266  packet[PKT_LENGTH_H] = MCY_HIBYTE(packet_length_out);
267 }

269 int Protocol2PacketHandler::txPacket(PortHandler *port, uint8_t *txpacket)
270 {
271  uint16_t total_packet_length = 0;
272  uint16_t written_packet_length = 0;
273 
274  if (port->is_using_)
275  return COMM_PORT_BUSY;
276  port->is_using_ = true;
277 
278  // byte stuffing for header
279  addStuffing(txpacket);
280 
281  // check max packet length
282  total_packet_length = MCY_MAKEWORD(txpacket[PKT_LENGTH_L], txpacket[PKT_LENGTH_H]) + 7;
283  // 7: HEADER0 HEADER1 HEADER2 RESERVED ID LENGTH_L LENGTH_H
284  if (total_packet_length > TXPACKET_MAX_LEN)
285  {
286  port->is_using_ = false;
287  return COMM_TX_ERROR;
288  }
289 
290  // make packet header
291  txpacket[PKT_HEADER0] = 0xFF;
292  txpacket[PKT_HEADER1] = 0xFF;
293  txpacket[PKT_HEADER2] = 0xFD;
294  txpacket[PKT_RESERVED] = 0x00;
295 
296  // add CRC16
297  uint16_t crc = updateCRC(0, txpacket, total_packet_length - 2); // 2: CRC16
298  txpacket[total_packet_length - 2] = MCY_LOBYTE(crc);
299  txpacket[total_packet_length - 1] = MCY_HIBYTE(crc);
300 
301  // tx packet
302  port->clearPort();
303  written_packet_length = port->writePort(txpacket, total_packet_length);
304  if (total_packet_length != written_packet_length)
305  {
306  port->is_using_ = false;
307  return COMM_TX_FAIL;
308  }
309 
310  return COMM_SUCCESS;
311 }

313 int Protocol2PacketHandler::rxPacket(PortHandler *port, uint8_t *rxpacket)
314 {
315  int result = COMM_TX_FAIL;
316 
317  uint16_t rx_length = 0;
318  uint16_t wait_length = 11; // minimum length (HEADER0 HEADER1 HEADER2 RESERVED ID LENGTH_L LENGTH_H INST ERROR CRC16_L CRC16_H)
319 
320  while(true)
321  {
322  rx_length += port->readPort(&rxpacket[rx_length], wait_length - rx_length);
323  if (rx_length >= wait_length)
324  {
325  uint16_t idx = 0;
326 
327  // find packet header
328  for (idx = 0; idx < (rx_length - 3); idx++)
329  {
330  if ((rxpacket[idx] == 0xFF) && (rxpacket[idx+1] == 0xFF) && (rxpacket[idx+2] == 0xFD) && (rxpacket[idx+3] != 0xFD))
331  break;
332  }
333 
334  if (idx == 0) // found at the beginning of the packet
335  {
336  if (rxpacket[PKT_RESERVED] != 0x00 ||
337  rxpacket[PKT_ID] > 0xFC ||
338  MCY_MAKEWORD(rxpacket[PKT_LENGTH_L], rxpacket[PKT_LENGTH_H]) > RXPACKET_MAX_LEN ||
339  rxpacket[PKT_INSTRUCTION] != 0x55)
340  {
341  // remove the first byte in the packet
342  for (uint16_t s = 0; s < rx_length - 1; s++)
343  rxpacket[s] = rxpacket[1 + s];
344  //memcpy(&rxpacket[0], &rxpacket[idx], rx_length - idx);
345  rx_length -= 1;
346  continue;
347  }
348 
349  // re-calculate the exact length of the rx packet
350  if (wait_length != MCY_MAKEWORD(rxpacket[PKT_LENGTH_L], rxpacket[PKT_LENGTH_H]) + PKT_LENGTH_H + 1)
351  {
352  wait_length = MCY_MAKEWORD(rxpacket[PKT_LENGTH_L], rxpacket[PKT_LENGTH_H]) + PKT_LENGTH_H + 1;
353  continue;
354  }
355 
356  if (rx_length < wait_length)
357  {
358  // check timeout
359  if (port->isPacketTimeout() == true)
360  {
361  if (rx_length == 0)
362  {
363  result = COMM_RX_TIMEOUT;
364  }
365  else
366  {
367  result = COMM_RX_CORRUPT;
368  }
369  break;
370  }
371  else
372  {
373  continue;
374  }
375  }
376 
377  // verify CRC16
378  uint16_t crc = MCY_MAKEWORD(rxpacket[wait_length-2], rxpacket[wait_length-1]);
379  if (updateCRC(0, rxpacket, wait_length - 2) == crc)
380  {
381  result = COMM_SUCCESS;
382  }
383  else
384  {
385  result = COMM_RX_CORRUPT;
386  }
387  break;
388  }
389  else
390  {
391  // remove unnecessary packets
392  for (uint16_t s = 0; s < rx_length - idx; s++)
393  rxpacket[s] = rxpacket[idx + s];
394  //memcpy(&rxpacket[0], &rxpacket[idx], rx_length - idx);
395  rx_length -= idx;
396  }
397  }
398  else
399  {
400  // check timeout
401  if (port->isPacketTimeout() == true)
402  {
403  if (rx_length == 0)
404  {
405  result = COMM_RX_TIMEOUT;
406  }
407  else
408  {
409  result = COMM_RX_CORRUPT;
410  }
411  break;
412  }
413  }
414 #if defined(__linux__) || defined(__APPLE__)
415  usleep(0);
416 #elif defined(_WIN32) || defined(_WIN64)
417  Sleep(0);
418 #endif
419  }
420  port->is_using_ = false;
421 
422  if (result == COMM_SUCCESS)
423  removeStuffing(rxpacket);
424 
425  return result;
426 }

432 
433  // tx packet
434  result = txPacket(port, txpacket);
435  if (result != COMM_SUCCESS)
436  return result;
437 
438  // // (Instruction == BulkRead or SyncRead) == this function is not available.
439  // if (txpacket[PKT_INSTRUCTION] == INST_BULK_READ || txpacket[PKT_INSTRUCTION] == INST_SYNC_READ)
440  // result = COMM_NOT_AVAILABLE;
441 
442  // (ID == Broadcast ID) == no need to wait for status packet or not available.
443  // (Instruction == action) == no need to wait for status packet
444  if (txpacket[PKT_ID] == BROADCAST_ID || txpacket[PKT_INSTRUCTION] == INST_ACTION)
445  {
446  port->is_using_ = false;
447  return result;
448  }
449 
450  // set packet timeout
451  if (txpacket[PKT_INSTRUCTION] == INST_READ)
452  {
453  port->setPacketTimeout((uint16_t)(MCY_MAKEWORD(txpacket[PKT_PARAMETER0+2], txpacket[PKT_PARAMETER0+3]) + 11));
454  }
455  else
456  {
457  port->setPacketTimeout((uint16_t)11);
458  // HEADER0 HEADER1 HEADER2 RESERVED ID LENGTH_L LENGTH_H INST ERROR CRC16_L CRC16_H
459  }
460 
461  // rx packet
462  do {
463  result = rxPacket(port, rxpacket);
464  } while (result == COMM_SUCCESS && txpacket[PKT_ID] != rxpacket[PKT_ID]);
465 
466  if (result == COMM_SUCCESS && txpacket[PKT_ID] == rxpacket[PKT_ID])
467  {
468  if (error != 0)
469  *error = (uint8_t)rxpacket[PKT_ERROR];
470  }
471 
472  return result;
473 }

502 int Protocol2PacketHandler::broadcastPing(PortHandler *port, std::vector<uint8_t> &id_list)
503 {

506

510  uint16_t wait_length = STATUS_LENGTH * MAX_ID;
511 
512  uint8_t txpacket[10] = {0};
513  uint8_t rxpacket[STATUS_LENGTH * MAX_ID] = {0};
514 
515  double tx_time_per_byte = (1000.0 / (double)port->getBaudRate()) * 10.0;
516 
517  txpacket[PKT_ID] = BROADCAST_ID;
518  txpacket[PKT_LENGTH_L] = 3;
519  txpacket[PKT_LENGTH_H] = 0;
520  txpacket[PKT_INSTRUCTION] = INST_PING;
521 
522  result = txPacket(port, txpacket);
523  if (result != COMM_SUCCESS)
524  {
525  port->is_using_ = false;
526  return result;
527  }
528 
529  // set rx timeout
530  //port->setPacketTimeout((uint16_t)(wait_length * 30));
531  port->setPacketTimeout(((double)wait_length * tx_time_per_byte) + (3.0 * (double)MAX_ID) + 16.0);
532 
533  while(1)
534  {
535  rx_length += port->readPort(&rxpacket[rx_length], wait_length - rx_length);
536  if (port->isPacketTimeout() == true)// || rx_length >= wait_length)
537  break;
538  }
539 
540  port->is_using_ = false;
541 
542  if (rx_length == 0)
543  return COMM_RX_TIMEOUT;
544 
545  while(1)
546  {
547  if (rx_length < STATUS_LENGTH)
548  return COMM_RX_CORRUPT;
549 
550  uint16_t idx = 0;
551 
552  // find packet header
553  for (idx = 0; idx < (rx_length - 2); idx++)
554  {
555  if (rxpacket[idx] == 0xFF && rxpacket[idx+1] == 0xFF && rxpacket[idx+2] == 0xFD)
556  break;
557  }
558 
559  if (idx == 0) // found at the beginning of the packet
560  {
561  // verify CRC16
562  uint16_t crc = MCY_MAKEWORD(rxpacket[STATUS_LENGTH-2], rxpacket[STATUS_LENGTH-1]);
563 
564  if (updateCRC(0, rxpacket, STATUS_LENGTH - 2) == crc)
565  {
566  result = COMM_SUCCESS;
567 
568  id_list.push_back(rxpacket[PKT_ID]);
569 
570  for (uint16_t s = 0; s < rx_length - STATUS_LENGTH; s++)
571  rxpacket[s] = rxpacket[STATUS_LENGTH + s];
572  rx_length -= STATUS_LENGTH;
573 
574  if (rx_length == 0)
575  return result;
576  }
577  else
578  {
579  result = COMM_RX_CORRUPT;
